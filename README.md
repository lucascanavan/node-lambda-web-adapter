# Sample AWS Lambda using nodejs, typescript, terraform & aws-lambda-web-adapter.

# Development
npm run dev

# Deploy ("terraform init" from "./terraform" will need to be run once before the first deploy)
npm run deploy

# Destroy (clean up)
npm run destroy
